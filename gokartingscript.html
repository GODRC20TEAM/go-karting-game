<body>

    <div id="game-container">

        <h1>Go Karting Game</h1>

        <p id="card-info">Card: None</p>

        <p id="points">Points: 0</p>

        <p id="lives">Lives: 5</p>

        <button onclick="playGame()">Play</button>

        <p id="status"></p>

        <button onclick="useSilverCard()">Use Silver Go Card</button>

        <button onclick="useGoldenCard()">Use Golden Go Card</button>

        <button onclick="refillLives()">Refill Lives (3h wait)</button>

    </div>

    <div id="leaderboard">

        <h2>Leaderboard</h2>

        <ol id="leaderboard-list"></ol>

    </div>

    <script>

        let lives = 5;

        let points = 0;

        let card = null;

        let leaderboard = [];

        function playGame() {

            if (lives > 0) {

                let score = Math.floor(Math.random() * 100);

                points += card === 'golden' ? score * 2 : score;

                lives--;

                document.getElementById('points').innerText = 'Points: ' + points;

                document.getElementById('lives').innerText = 'Lives: ' + lives;

                document.getElementById('status').innerText = 'You earned ' + score + ' points!';

               

                updateLeaderboard();

            } else {

                document.getElementById('status').innerText = 'No lives left! Please refill.';

            }

        }

        function useSilverCard() {

            card = 'silver';

            document.getElementById('card-info').innerText = 'Card: Silver Go Card';

        }

        function useGoldenCard() {

            card = 'golden';

            document.getElementById('card-info').innerText = 'Card: Golden Go Card';

        }

        function refillLives() {

            setTimeout(() => {

                lives = 5;

                document.getElementById('lives').innerText = 'Lives: ' + lives;

            }, 10800000); // 3 hours in milliseconds

        }

        function updateLeaderboard() {

            leaderboard.push(points);

            leaderboard.sort((a, b) => b - a);

            let list = document.getElementById('leaderboard-list');

            list.innerHTML = '';

            for (let i = 0; i < leaderboard.length && i < 10; i++) {

                let listItem = document.createElement('li');

                listItem.innerText = leaderboard[i];

                list.appendChild(listItem);

            }

        }

    </script>
